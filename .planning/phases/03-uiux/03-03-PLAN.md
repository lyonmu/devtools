    3. In `render_tab_content` for the Cert branch (active_menu == 0), add GPUI drag-drop event handlers to the cert tab content div:

    ```rust
    // On the cert tab content div:
    .on_drag_enter(cx.listener(|this, _, _, cx| {
        if let DevToolsApp::Cert(t) = this {
            t.drag_hover = true;
            cx.notify();
        }
    }))
    .on_drag_leave(cx.listener(|this, _, _, cx| {
        if let DevToolsApp::Cert(t) = this {
            t.drag_hover = false;
            cx.notify();
        }
    }))
    .on_drop(cx.listener(|this, paths: Vec<std::path::PathBuf>, _window, cx| {
        if let DevToolsApp::Cert(t) = this {
            t.drag_hover = false;
            // Find first supported file
            let supported_exts = ["pem", "der", "p12", "pfx", "cer", "crt"];
            let file = paths.iter().find(|p| {
                p.extension().map(|e| supported_exts.contains(&e.to_str().unwrap_or(""))).unwrap_or(false)
            });
            match file {
                Some(path) => {
                    t.is_importing = true;
                    t.import_error = None;
                    cx.notify();
                    // Parse file (reuse existing detect_and_parse)
                    let result = crate::cert::detect_and_parse(path);
                    t.is_importing = false;
                    match result {
                        Ok(certs) => {
                            if let Some(first) = certs.into_iter().next() {
                                t.loaded_cert = Some(first);
                                t.import_error = None;
                            }
                        }
                        Err(e) => {
                            t.import_error = Some(e);
                        }
                    }
                    cx.notify();
                }
                None => {
                    t.import_error = Some("不支持的文件格式。支持: .pem, .der, .p12, .pfx, .cer, .crt".to_string());
                    cx.notify();
                }
            }
        }
    }))
    ```

    Note: GPUI's drag-drop API may differ from this pseudocode. Check GPUI 0.2 docs for exact event names.
    If GPUI doesn't have native drag-drop, implement via platform-specific integration or skip drag-drop and keep button-only import.
