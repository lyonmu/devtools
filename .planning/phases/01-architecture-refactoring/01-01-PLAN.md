pub fn render_status_banner(kind: UiStatusKind, message: impl Into<String>) -> gpui::Div {
    let (border, text) = match kind {
        UiStatusKind::Empty => (rgb(0x888899), rgb(0xaaaabb)),
        UiStatusKind::Success => (rgb(0x22c55e), rgb(0xbbf7d0)),
        UiStatusKind::Error => (rgb(0xf87171), rgb(0xfecaca)),
        UiStatusKind::Info => (rgb(0x3b82f6), rgb(0xbfdbfe)),
    };
    div()
        .w_full().px_3().py_2()
        .bg(rgb(0x1e1e2e))
        .border_l_4().border_color(border)
        .rounded_md()
        .text_size(FONT_BODY)
        .text_color(text)
        .child(message.into())
}
```

4. **render_mono_output_block** function — exact copy from app.rs lines 144-157, as `pub fn`:

pub fn render_mono_output_block(text: &str) -> gpui::Stateful<gpui::Div> {
    div()
        .w_full().px_3().py_2()
        .bg(rgb(0x1a1a2a))
        .rounded_md()
        .id(gpui::ElementId::Name(gpui::SharedString::from(format!("mono-output-{}", text.len()))))
        .overflow_x_scroll()
        .font_family("monospace")
        .text_size(FONT_SMALL)
        .text_color(rgb(0xddddcc))
        .child(text.to_string())
}
```
